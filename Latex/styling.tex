\newcommand{\bpxtable}[3]
{
    \begin{center}
        \rowcolors{2}
        {red!15}
        {blue!15}
        \begin{tabulary}{\linewidth}{|#1|}
            \hline
            \bpxtabhead{#2}

            \hline\hline
            #3
            \hline
        \end{tabulary}
    \end{center}
}

{
